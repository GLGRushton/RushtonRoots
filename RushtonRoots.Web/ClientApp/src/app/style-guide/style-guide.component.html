color_lens Angular Material Theme Customization

Complete guide to customizing the Angular Material theme for RushtonRoots.

Theme Configuration Custom Material theme with RushtonRoots colors

The application uses a custom Angular Material theme defined in styles.scss: 

@use '@angular/material' as mat;

@include mat.core();

$rushton-primary: mat.define-palette(mat.$green-palette, 700, 500, 900);
$rushton-accent: mat.define-palette(mat.$light-green-palette, 400, 300, 600);
$rushton-warn: mat.define-palette(mat.$red-palette);

$rushton-theme: mat.define-light-theme((
  color: (
    primary: $rushton-primary,
    accent: $rushton-accent,
    warn: $rushton-warn,
  )
));

@include mat.all-component-themes($rushton-theme);

 Reactive Forms Pattern Standard pattern for form components 
import { FormBuilder, FormGroup, Validators } from '@angular/forms';

export class MyFormComponent {
  form: FormGroup;

  constructor(private fb: FormBuilder) {
    this.form = this.fb.group({
      firstName: ['', [Validators.required, Validators.maxLength(100)]],
      lastName: ['', [Validators.required, Validators.maxLength(100)]],
      email: ['', [Validators.required, Validators.email]]
    });
  }

  onSubmit(): void {
    if (this.form.valid) {
      // Handle form submission
    }
  }
}
 Accessibility Best Practices WCAG 2.1 AA compliance guidelines
  • Always provide ARIA labels for interactive elements without visible text
  • Use semantic HTML (buttons for actions, links for navigation)
  • Ensure color contrast meets 4.5:1 minimum for normal text
  • Make all functionality keyboard accessible
  • Provide alt text for images
  • Use proper heading hierarchy (h1, h2, h3, etc.)
  • Test with screen readers (NVDA, JAWS, VoiceOver)
