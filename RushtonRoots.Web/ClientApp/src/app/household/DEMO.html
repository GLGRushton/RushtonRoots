Usage in Razor Views

@model HouseholdIndexViewModel

<div class="container">
    <app-household-index
        initial-households='@Json.Serialize(Model.Households)'
        can-edit="@(User.IsInRole("Admin") ? "true" : "false")"
        can-delete="@(User.IsInRole("Admin") ? "true" : "false")"
        can-create="true">
    </app-household-index>
</div>
